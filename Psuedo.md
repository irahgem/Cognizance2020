# Psuedo Code
<!-- OL -->
 1. Start
 2. Read Number
 3. Initialize Check = Number
 4. Initialize Sum = 0
 5. while Check!=0 <br>
      Rem = check%10 <br>
      Sum = Sum + (Rem * Rem * Rem) <br>
      Check = Check/10 
 6. if (Sum==Number) <br>
      write"Given number is Armstrong" <br>
      else <br>
     write"Given Number is not Armstrong" <br>
 7. Stop
